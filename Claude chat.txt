# Helper: capture print output so we can assert on it
from io import StringIO
import sys

def run_and_capture(func, *args):
    buf = StringIO()
    sys.stdout = buf
    func(*args)
    sys.stdout = sys.__stdout__
    return buf.getvalue()

class TestLateSettlements(unittest.TestCase):

        """Transaction in March settled in April should be flagged."""
        txns = [
            {"txn_id": "T1", "date": date(2024, 3, 28), "amount": Decimal("100"), "status": "completed"}
        ]
        sets = [
            {"settlement_id": "S1", "txn_id": "T1", "settlement_date": date(2024, 4, 2), "settled_amount": Decimal("100")}
        ]

        self.assertIn("Late Settlements Detected!", output)

    def test_same_month_settlement_is_clean(self):

Reconciliation System — Simple Explanation
1. What Problem It Solves
Payment companies process thousands of transactions daily. A separate system (the bank or payment processor) settles those transactions — transferring the actual money. These two records should match perfectly, but in practice they don't always.

This system compares the company's internal transaction log against the settlement records to find where the money doesn't line up. Without this, discrepancies go unnoticed, leading to financial loss, duplicate payouts, or unresolved refunds.

2. How the Datasets Are Structured
Transactions — what the company recorded happening:


txn_id | date       | amount  | merchant    | status
TXN001 | 2024-03-01 | $150.00 | ShopMart    | completed
TXN015 | 2024-03-17 | -$99.00 | ShopMart    | refunded
Settlements — what the bank confirmed was paid out:


settlement_id | txn_id | settlement_date | settled_amount
SET001        | TXN001 | 2024-03-03      | $150.00
SET016        | TXN016 | 2024-03-10      | $410.00   ← appears twice
Each settlement row is supposed to map 1-to-1 with a transaction. The reconciliation system joins them on txn_id and checks for breaks.

3. What Mismatches It Detects
Gap	Function	What it checks
Late settlement	detect_late_settlements	Transaction month ≠ settlement month. Money moved in a different accounting period.
Rounding difference	detect_rounding_differences	Settled amount differs from transaction amount by a tiny value (≤ $0.01). Individually dismissible, but accumulates across records.
Duplicate settlement	detect_duplicate_settlements	Same txn_id + date + amount appears more than once in settlements. Risk of paying twice.
Orphan refund	detect_orphan_refunds	A refund references an original transaction ID that doesn't exist. Money refunded with no traceable purchase.
4. Example Output Explanation

Late Settlements Detected!
  - TXN011 transacted on 2024-03-28 but settled on 2024-04-02
TXN011 happened in March but the bank paid it in April — it will appear in March's books as pending but won't close until April's reconciliation runs.


Rounding Differences Detected!
  - TXN012 | transaction=33.335, settled=33.34, per-record diff=0.005
  - AGGREGATE drift from rounding = 0.015
Each individual record is off by half a cent — easy to ignore. But across many records it becomes $0.015 here, potentially thousands of dollars at scale.


Duplicate Settlements Detected!
  - TXN016 settled twice — SET016 and SET016B both on 2024-03-10 for 410.00
The merchant was paid $410 twice. One of these needs to be reversed.


Orphan Refunds Detected!
  - TXN015 is a refund for TXN999 but TXN999 does not exist in transactions
A $99 refund was issued, but there's no record of the original purchase it came from — either the original was deleted, mis-keyed, or this is a fraudulent refund.

3 Realistic Limitations in Production
1. No timezone handling.
Transactions recorded near midnight can shift date by a day depending on the server's timezone vs the bank's timezone. A transaction at 11:58 PM EST becomes next day in UTC — falsely flagged as a late settlement.

2. Partial settlements not supported.
Some payment processors settle one transaction in multiple partial batches (e.g. $500 split into $300 + $200). The current system matches on txn_id one-to-one and would misread this as a rounding error or missing settlement.

3. Tolerance is hardcoded, not currency-aware.
The rounding tolerance is fixed at $0.01 USD. For JPY (no decimals) this threshold is wrong, and for high-value transactions in any currency, a 1-cent tolerance may mask a real discrepancy or flag a legitimate one.

        print("Late Settlements Detected!")
        for msg in issues:
            print(f"  - {msg}")
    else:
        print("Late Settlements Detected! : None")


def detect_rounding_differences(transactions, settlements, tolerance=Decimal("0.01")):
    """GAP B — rounding differences visible only when totals are summed.

    Individual per-txn diffs may each be within tolerance (so they look fine),
    but the aggregate difference across all matched pairs reveals the gap.
    """
    settlement_map = {}
    for s in settlements:
        settlement_map.setdefault(s["txn_id"], Decimal("0"))
        settlement_map[s["txn_id"]] += s["settled_amount"]

    txn_map = {}
    for txn in transactions:
        txn_map.setdefault(txn["txn_id"], Decimal("0"))
        txn_map[txn["txn_id"]] += txn["amount"]

    issues = []

    for txn_id, txn_amount in txn_map.items():
        if txn_id in settlement_map:
            per_diff = abs(settlement_map[txn_id] - txn_amount)
            if Decimal("0") < per_diff <= tolerance:
                issues.append(
                    f"{txn_id} | transaction={txn_amount}, "
                    f"settled={settlement_map[txn_id]}, per-record diff={per_diff}"
                )

    if issues:
        rounding_diffs = [
            abs(settlement_map[txn_id] - amount)
            for txn_id, amount in txn_map.items()
            if txn_id in settlement_map
            and Decimal("0") < abs(settlement_map[txn_id] - amount) <= tolerance
        ]
        total_drift = sum(rounding_diffs)
        issues.append(
            f"AGGREGATE drift from rounding = {total_drift} "
            f"(each diff {rounding_diffs} looks small, but they accumulate)"
        )

    if issues:
        print("Rounding Differences Detected!")
        for msg in issues:
            print(f"  - {msg}")
    else:
        print("Rounding Differences Detected! : None")


def detect_duplicate_settlements(settlements):
    """GAP D — same txn_id settled more than once with the same amount and date."""
    seen = {}
    issues = []

    for s in settlements:
        key = (s["txn_id"], s["settlement_date"], s["settled_amount"])
        if key in seen:
            issues.append(
                f"{s['txn_id']} settled twice — "
                f"{seen[key]} and {s['settlement_id']} "
                f"both on {s['settlement_date']} for {s['settled_amount']}"
            )
        else:
            seen[key] = s["settlement_id"]

    if issues:
        print("Duplicate Settlements Detected!")
        for msg in issues:
            print(f"  - {msg}")
    else:
        print("Duplicate Settlements Detected! : None")


def detect_orphan_refunds(transactions):
    """GAP C — refund transaction references an original txn_id that does not exist."""
    txn_ids = {txn["txn_id"] for txn in transactions}
    issues = []

    for txn in transactions:
        if txn["status"] == "refunded":
            original = txn.get("refund_for")
            if original and original not in txn_ids:
                issues.append(
                    f"{txn['txn_id']} is a refund for {original} "
                    f"but {original} does not exist in transactions"
                )

    if issues:
        print("Orphan Refunds Detected!")
        for msg in issues:
            print(f"  - {msg}")
    else:
        print("Orphan Refunds Detected! : None")
